#include <stdio.h>
 
int main() {
    int x,cnt=0;
    scanf("%d",&x);
    while (1) {
        if (x%5==0) {
            cnt = cnt + x/5;
            break;
        }
        x = x - 3;
        cnt++;
        if (x <= 0) break;
    }
    if (x < 0) printf("-1\n");
    else printf("%d\n", cnt);
}
